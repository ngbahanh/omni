### 1B. i18n engine — src/i18n/index.js

```js
// src/i18n/index.js
import { loadConfig } from '../config.js';

let currentLocale = 'en';
let strings = {};

export async function initI18n() {
  const config = loadConfig('omni-global', { language: 'en' });
  currentLocale = config.language ?? 'en';
  const mod = await import(`./locales/${currentLocale}.js`);
  strings = mod.default;
}

// t('key') hoặc t('key', { name: 'Nam' }) cho interpolation
export function t(key, vars = {}) {
  const parts = key.split('.');
  let val = strings;
  for (const p of parts) {
    val = val?.[p];
    if (val === undefined) return key; // fallback: trả về key nếu không tìm thấy
  }
  if (typeof val !== 'string') return key;
  return val.replace(/\{\{(\w+)\}\}/g, (_, k) => vars[k] ?? `{{${k}}}`);
}

export function getCurrentLocale() { return currentLocale; }

export function getSupportedLocales() {
  return [
    { code: 'en', label: 'English' },
    { code: 'vi', label: 'Tiếng Việt' },
  ];
}
````

Gọi `await initI18n()` trong `src/index.js` TRƯỚC khi làm bất kỳ thứ gì khác.

### 1D. Quy tắc sử dụng i18n trong toàn project

- Mọi chuỗi hiển thị cho user PHẢI đi qua `t('key')` — không được hardcode text trong code.
- Import `t` từ `../i18n/index.js` trong bất kỳ file nào cần hiển thị text.
- Khi thêm string mới: thêm vào CẢ HAI file en.js và vi.js cùng lúc — không để thiếu key.
- Nếu một key tồn tại trong en.js nhưng thiếu trong vi.js → engine tự động fallback về key string (hiển thị key thay vì crash).
- Namespace theo cấu trúc: `<section>.<key>` — ví dụ `settings.auth_title`, `common.cancel`.
- Interpolation dùng `{{varName}}` syntax: `t('first_run.step', { current: 1, total: 4 })`.
- Khi thêm tool mới vào omni → thêm namespace mới vào locale files cho tool đó.

## PHẦN 2 — FIRST-RUN SETUP (src/firstRun.js)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

### 2A. Khi nào chạy First-run

Trong `src/index.js`, ngay sau `initI18n()`, kiểm tra:

```js
import { isFirstRun, runFirstRunSetup } from "./firstRun.js";

async function main() {
  await initI18n(); // load ngôn ngữ (mặc định 'en' nếu chưa có config)
  banner();

  if (await isFirstRun()) {
    await runFirstRunSetup();
    await initI18n(); // reload i18n sau khi user chọn ngôn ngữ
  }

  await showMenu();
}
```

`isFirstRun()` → đọc config `omni-global`, trả về `true` nếu chưa có field `setupCompleted: true`.

First-run chỉ chạy MỘT LẦN DUY NHẤT. Sau khi hoàn tất, lưu `setupCompleted: true` vào config.

## PHẦN 4 — TÍCH HỢP VÀO src/index.js

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Cập nhật `src/index.js` theo đúng thứ tự này:

```js
#!/usr/bin/env node
import { initI18n } from "./i18n/index.js";
import { isFirstRun, runFirstRunSetup } from "./firstRun.js";
import { initTheme, banner } from "./ui.js";
import { showMenu } from "./menu.js";

async function main() {
  await initI18n(); // 1. load language từ config (default 'en')
  await initTheme(); // 2. load theme từ config (default 'default')
  banner(); // 3. hiển thị banner với theme đã load

  if (await isFirstRun()) {
    await runFirstRunSetup();
    await initI18n(); // 4. reload i18n vì user vừa chọn ngôn ngữ
    await initTheme(); // 5. reload theme vì user vừa chọn theme
  }

  await showMenu(); // 6. vào menu chính
}

main().catch((err) => {
  console.error("Fatal error:", err.message);
  process.exit(1);
});
```

Cập nhật `src/ui.js` để có hàm `initTheme()`:

- Đọc theme từ config `omni-global`
- Set biến global theme trong module
- Các hàm `ui.*` dùng theme đó để chọn màu chalk

## PHẦN 7 — YÊU CẦU KỸ THUẬT

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

- i18n engine phải là pure ESM module, không side effects khi import
- `t()` phải là synchronous sau khi `initI18n()` đã chạy
- Không dùng thư viện i18n bên ngoài (tự build theo spec) — giữ dependencies tối giản
- First-run và Settings đều dùng cùng một config file `omni-global.json`
- Settings tool phải loop (quay lại menu settings sau mỗi action) cho đến khi user chọn "Back to main"
- Tất cả input prompts dùng t() cho message — không hardcode
- Theme phải áp dụng ngay lập tức khi thay đổi, không cần restart
- Tất cả wrap trong try/catch, dùng logger từ src/logger.js
